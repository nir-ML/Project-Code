\subsection{Therapeutic Similarity Calculation}

The therapeutic similarity score is computed as a weighted combination of four sub-scores:

\begin{equation}
T(d_{\text{alt}}) = \alpha_1 \cdot \text{ATC}(d_{\text{alt}}) + \alpha_2 \cdot \text{Target}(d_{\text{alt}}) + \alpha_3 \cdot \text{Disease}(d_{\text{alt}}) + \alpha_4 \cdot \text{Pathway}(d_{\text{alt}})
\end{equation}

where $\alpha_1 = 0.35$, $\alpha_2 = 0.25$, $\alpha_3 = 0.25$, $\alpha_4 = 0.15$.

\subsubsection{ATC Code Matching}
\begin{equation}
\text{ATC}(d_{\text{alt}}) = \begin{cases}
1.0 & \text{if same 5th level ATC code} \\
0.8 & \text{if same 4th level ATC code} \\
0.5 & \text{if same 3rd level ATC code} \\
0.2 & \text{if same 2nd level ATC code} \\
0.0 & \text{otherwise}
\end{cases}
\end{equation}

\subsubsection{Protein Target Overlap}
\begin{equation}
\text{Target}(d_{\text{alt}}) = \frac{|T_{d_{\text{orig}}} \cap T_{d_{\text{alt}}}|}{|T_{d_{\text{orig}}} \cup T_{d_{\text{alt}}}|}
\end{equation}

\subsubsection{Disease Indication Overlap}
\begin{equation}
\text{Disease}(d_{\text{alt}}) = \frac{|D_{d_{\text{orig}}} \cap D_{d_{\text{alt}}}|}{|D_{d_{\text{orig}}}|}
\end{equation}

\subsection{Safety Improvement Formula}
\begin{equation}
\text{SAF}(d_{\text{alt}}) = 1 - \frac{\sum_{j \in R} w_{\text{sev}(d_{\text{alt}}, j)}}{\sum_{j \in R} w_{\text{sev}(d_{\text{orig}}, j)}}
\end{equation}

where $R$ is the set of drugs in the current regimen and $w_{\text{sev}}$ is the severity weight from Table~\ref{tab:safety_penalties}.

\section{Algorithm Pseudocode}

\begin{verbatim}
Algorithm: Multi-Objective Drug Recommendation

Input: 
  - current_regimen: Set of drugs
  - target_drug: Drug to be replaced
  - kg: Knowledge graph

Output:
  - ranked_alternatives: List of (drug, score) tuples

1. candidates = find_therapeutic_alternatives(target_drug, kg)
2. for each candidate in candidates:
3.     T = compute_therapeutic_similarity(candidate, target_drug, kg)
4.     SAF = compute_safety_improvement(candidate, target_drug, 
            current_regimen, kg)
5.     R = compute_risk_reduction(candidate, target_drug, 
            current_regimen, kg)
6.     score = 0.40*T + 0.35*SAF + 0.25*R
7.     add (candidate, score) to ranked_alternatives
8. sort ranked_alternatives by score descending
9. return ranked_alternatives
\end{verbatim}
