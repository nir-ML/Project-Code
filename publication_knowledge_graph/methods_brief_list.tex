\subsection*{1. Knowledge Graph Construction}
% ============================================================================

\textbf{Data Sources:}
\begin{enumerate}[label=(\roman*)]
    \item \textbf{DrugBank v5.1.9} -- Primary source for drug identifiers, DDI descriptions, protein targets, pathways, and pharmacogenomic data (1.8 GB XML)
    \item \textbf{SIDER 4.1} -- Drug-side effect associations using MedDRA terminology (1,110 drugs matched; 265,238 associations)
    \item \textbf{CTD} -- Drug-disease relationships via CAS number matching (52,753) and name matching (10,525)
\end{enumerate}

\textbf{Fact-Based Integration Strategy:}
\begin{enumerate}[label=(\roman*)]
    \item DrugBank ID matching (primary)
    \item CAS number matching (secondary)
    \item Exact drug name matching, case-insensitive (tertiary)
\end{enumerate}

\textbf{Graph Statistics:}
\begin{itemize}
    \item Total Nodes: 45,655 (Drugs: 4,313; Proteins: 3,176; Side Effects: 5,548; Diseases: 3,041; Pathways: 25,958; Categories: 3,619)
    \item Total Edges: 1,211,674 (DDIs: 759,774; Drug-Side Effect: 265,238; Drug-Category: 70,618; Drug-Disease: 63,278; Drug-Pathway: 31,207; Drug-Protein: 21,559)
\end{itemize}

\subsection*{2. DDI Severity Classification}
% ============================================================================

\textbf{Methodology:}
\begin{enumerate}[label=(\roman*)]
    \item Rule-based classifier using keyword weights derived from DDInter database (n=26,014)
    \item Scoring function: $\text{score}(d) = \sum_{k \in \mathcal{K}} w_k \cdot \mathbb{1}[k \in d]$
    \item Log-likelihood weights: $w_k = \ln(P(k|\text{Major})/P(k|\text{Moderate}))$
\end{enumerate}

\textbf{Severity Thresholds (Percentile-Based):}
\begin{itemize}
    \item Contraindicated: score $\geq$ P96 (top 4\%)
    \item Major: score $\geq$ P92 (top 8\%)
    \item Minor: score $\leq$ P2 (bottom 2\%)
    \item Moderate: Remaining interactions
\end{itemize}

\textbf{Validation:}
\begin{itemize}
    \item DDInter benchmark: 66.4\% exact accuracy
    \item Cohen's $\kappa$ = +0.096 ($p < 0.0001$)
\end{itemize}

\textbf{Final Distribution:}
\begin{itemize}
    \item Moderate: 80.1\%
    \item Minor: 9.3\%
    \item Contraindicated: 5.8\%
    \item Major: 4.7\%
\end{itemize}

\subsection*{3. Polypharmacy Risk Index (PRI)}
% ============================================================================

\textbf{Definition:}
\begin{equation}
\text{PRI}(d) = 0.25 \cdot C_{\text{degree}}(d) + 0.30 \cdot C_{\text{weighted}}(d) + 0.20 \cdot C_{\text{betweenness}}(d) + 0.25 \cdot S(d)
\end{equation}

\textbf{Components:}
\begin{enumerate}[label=(\roman*)]
    \item $C_{\text{degree}}(d)$: Degree centrality -- Number of interactions (weight: 0.25)
    \item $C_{\text{weighted}}(d)$: Weighted degree -- Severity-weighted interaction sum (weight: 0.30)
    \item $C_{\text{betweenness}}(d)$: Betweenness centrality -- Risk propagation potential (weight: 0.20)
    \item $S(d)$: Severity profile -- Proportion of contraindicated/major interactions (weight: 0.25)
\end{enumerate}

\textbf{Interpretation:}
\begin{itemize}
    \item PRI $>$ 0.5: High risk -- requires immediate review
    \item PRI 0.3--0.5: Medium risk -- monitor closely
    \item PRI $<$ 0.3: Low risk -- standard monitoring
\end{itemize}

% ============================================================================
\subsection*{4. Multi-Objective Drug Recommendation System}
% ============================================================================

\textbf{Recommendation Score Formula:}
\begin{equation}
\text{RecScore}(d_{\text{alt}}) = 0.40 \cdot T(d_{\text{alt}}) + 0.35 \cdot SAF(d_{\text{alt}}) + 0.25 \cdot R(d_{\text{alt}})
\end{equation}

\textbf{Scoring Components:}
\begin{enumerate}[label=(\roman*)]
    \item \textbf{Therapeutic Similarity} $T(d)$ (weight: 0.40):
    \begin{itemize}
        \item ATC code matching (pharmacological class)
        \item Shared protein targets (mechanism of action)
        \item Disease indication overlap
        \item Metabolic pathway similarity
    \end{itemize}
    
    \item \textbf{Safety Improvement} $SAF(d)$ (weight: 0.35):
    \begin{itemize}
        \item DDI count reduction with current regimen
        \item Contraindicated interaction avoidance
        \item Major interaction reduction
    \end{itemize}
    
    \item \textbf{Risk Reduction} $R(d)$ (weight: 0.25):
    \begin{itemize}
        \item Net change in severe interactions (Contraindicated + Major)
        \item PRI score improvement
    \end{itemize}
\end{enumerate}

\subsection*{5. Case Study: Cardiovascular Polypharmacy}
% ============================================================================

\textbf{High-Risk Regimen (5 drugs):}
\begin{enumerate}[label=(\roman*)]
    \item Warfarin (anticoagulant)
    \item Amiodarone (antiarrhythmic)
    \item Digoxin (cardiac glycoside)
    \item Quinidine (antiarrhythmic)
    \item Propranolol (beta-blocker)
\end{enumerate}

\textbf{Verified DDI Interactions:}
\begin{itemize}
    \item \textbf{Contraindicated (2):} Warfarin+Amiodarone (bleeding risk), Digoxin+Propranolol (bradycardia)
    \item \textbf{Major (8):} Warfarin+Digoxin, Warfarin+Quinidine, Warfarin+Propranolol, Amiodarone+Digoxin, Amiodarone+Quinidine, Amiodarone+Propranolol, Digoxin+Quinidine, Quinidine+Propranolol
\end{itemize}

\textbf{Recommendation Result:}
\begin{itemize}
    \item \textbf{Substitute:} Warfarin $\rightarrow$ Dabigatran
    \item \textbf{Recommendation Score:} 0.801
    \begin{itemize}
        \item Therapeutic Similarity: 0.850 (40\%)
        \item Safety Improvement: 0.782 (35\%)
        \item Risk Reduction: 0.750 (25\%)
    \end{itemize}
\end{itemize}

\textbf{Risk Reduction Achieved:}
\begin{itemize}
    \item Contraindicated: 2 $\rightarrow$ 1 (--50\%)
    \item Major: 8 $\rightarrow$ 6 (--25\%)
    \item Total Severe: 10 $\rightarrow$ 7 (--30\%)
    \item Average PRI: 0.569 $\rightarrow$ 0.504 (--11.5\%)
\end{itemize}

\section*{Key Findings Summary}
% ============================================================================

\begin{enumerate}
    \item \textbf{Blood Class Highest Risk:} Blood $\leftrightarrow$ Musculoskeletal pairs show 87.3\% severe interactions
    \item \textbf{Volume-Severity Paradox:} High DDI counts $\neq$ high clinical risk (Cardiovascular-Nervous: 103,832 DDIs, only 5.3\% severe)
    \item \textbf{Multi-Objective Optimization:} Balancing therapeutic similarity (40\%), safety (35\%), and risk reduction (25\%) identifies clinically appropriate substitutes
    \item \textbf{Validated Case Study:} Dabigatran substitution achieves 30\% reduction in severe interactions while maintaining therapeutic equivalence
\end{enumerate}
